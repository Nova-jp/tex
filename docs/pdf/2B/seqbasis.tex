数列の基本形
\begin{enumerate}
    \item （等差型）$a_1=2, a_{n+1}=a_n +3$
    \item （等比型）$a_1=2, a_{n+1}=3 a_n$
    \item （階差型）$a_1=2, a_{n+1}=a_n -2n+5$
    \item （階比型）$a_1=2, a_{n+1}=a_n +3^n$
\end{enumerate}
以下は文字を置いて解くタイプ
\begin{enumerate}
    \item $a_1=1, a_{n+1}=2 a_n -3$
    \item $a_1=2, a_{n+1}=2 a_n -3n+4$
    \item $a_1=2, a_{n+1}=3a_n+n^2 -2n +5$
\end{enumerate}
